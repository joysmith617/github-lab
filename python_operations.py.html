#String Manipulation

#Prompt the user to put their first and last name
first_name = input("Enter your first name: ")
last_name = input("Enter your last name: ")

#concentate using strings to create full name
greeting = "Hello, " + first_name + " " + last_name + " Welcome to Python programming."

#print greeting
print(greeting)


#Convert input from the strings to numbers float for arithmetic operation
num1 = float(input("Enter the first number: "))
num2 = float(input("Enter the second number: "))

#Arithmetic Operations

#Perform Arithmetic Operations
addition = num1 + num2
subtraction = num1 - num2
multiplication = num1 * num2
integer_division = num1 // num2
floating_point_division = num1 / num2


#Step 4: Display the results
print(f"Addition: {num1} + {num2} = {addition}")
print(f"Subtraction: {num1} - {num2} = {subtraction}")
print(f"Multiplication: {num1} * {num2} = {multiplication}")
print(f"Integer Division: {num1} // {num2} = {integer_division}")
print(f"Floating-point Division: {num1} / {num2} = {floating_point_division}")


#Using the 'math' module

#Step 1: Import the math module
import math

#Step 2: Get user input and perform operations
number = float(input("\nEnter a number to calculate its square root: "))
square_root = math.sqrt(number)

base = float(input("Enter the base: "))
exponent = float(input("Enter the exponent: "))
power_result = math.pow(base, exponent)

pi_value = round(math.pi, 5)

#Step 3: Display Results
print(f"\nThe square root of {number} is {square_root}")
print(f"{base} raised to the power of {exponent} is {power_result}")
print(f"The value of pi rounded to 5 decimal places is {pi_value}")


#Variable Reassignment and Type Conversion

#Step 1: Assign an integer value to a variable
num = 10
print(f"\nOriginal integer value: {num} (type: {type(num)})")

#Step 2: Reassign the same variable with a floating-point number
num = 10.5
print(f"Reassigned floating-point value: {num} (type: {type(num)})")


# Step 3: Convert a string containing a number into an integer
str_number = "25"
converted_number = int(str_number)

# Step 4: Perform arithmetic with the converted number
sum_result = converted_number + 5

#Display Results
print(f"Converted string to integer: {converted_number} (type: {type(converted_number)})")
print(f"Sum of converted integer and 5: {sum_result}")
